✗ MySQL MET utf8 (BEPERKT)
mysql — gebruiker@localhost
-- Tabel aangemaakt met 'utf8' charset mysql> CREATE TABLE posts ( inhoud VARCHAR(255) ) CHARSET=utf8; mysql> INSERT INTO posts VALUES ('Top resultaat! 🚀'); ERROR 1366 (HY000): Incorrect string value: '\xF0\x9F\x9A\x80' for column 'inhoud' at row 1 -- utf8 in MySQL = max 3 bytes (U+0000–U+FFFF) -- Emoji 🚀 = U+1F680 = 4 bytes → GEBLOKKEERD
✓ MySQL MET utf8mb4 (VOLLEDIG UNICODE)
mysql — gebruiker@localhost
-- utf8mb4 = echte UTF-8 (max 4 bytes) mysql> ALTER TABLE posts CONVERT TO CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci; Query OK, 0 rows affected mysql> INSERT INTO posts VALUES ('Top resultaat! 🚀'); Query OK, 1 row affected mysql> SELECT inhoud FROM posts; Top resultaat! 🚀
MySQL's utf8 charset is een historische fout: het ondersteunt slechts 3 bytes per karakter (U+0000–U+FFFF). Emoji's en zeldzame Unicode-tekens vereisen 4 bytes. Gebruik altijd utf8mb4 voor echte UTF-8 ondersteuning. Dit geldt ook voor MariaDB.
